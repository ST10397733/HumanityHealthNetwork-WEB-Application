<script>
  // Simple demo user system for fake login
  const users = [{ username: "admin", password: "admin123" }, { username: "volunteer", password: "hhnetwork" }];

  function loginUser() {
    const uname = document.getElementById('login-username').value;
    const pwd = document.getElementById('login-password').value;
    const found = users.find(u => u.username === uname && u.password === pwd);
    if (found) {
      alert("Login successful. Welcome, " + uname + "!");
    } else {
      alert("Invalid credentials. Please try again.");
    }
    return false; // prevent actual form submission
  }

  function registerUser() {
    const email = document.getElementById('signup-email').value;
    const uname = document.getElementById('signup-username').value;
    const pwd = document.getElementById('signup-password').value;
    if (uname && pwd && email) {
      alert("Signup successful. Welcome, " + uname + "!");
    }
    return false; // prevent actual form submission
  }
</script>
